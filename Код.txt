#include <iostream>
#include <ctime>
using namespace std;
int main() {
	setlocale(LC_ALL, "Russian");
	srand(time(0));
	int N = 4, M = 3, min = 0;
	int **a = new int*[N];
	int **b = new int*[N];
	for (int i = 0; i != N; i++) {
		a[i] = new int[M];
	}
	for (int i = 0; i != N; i++) {
		b[i] = new int[M];
	}
	for (int i = 0; i != N; i++) {
		for (int j = 0; j != M; j++) {
			a[i][j] = rand() % 21 - 10;
			cout << a[i][j] << " ";
		}
		cout << endl;
	}
	for (int i = 0; i < N; ++i) {
		for (int j = 0; j < M; ++j) {
			if (a[i][j] > 0) {
				b[i][j] = a[i][j];
				min = b[0][j];
				if (b[i][j] <= min) {
					min = a[i][j];
				}
			}
		}
		cout << "min" << min << endl;
	}
}
